# Install Polybar
    $ sudo pacman -S polybar

# Run Polybar
    $ polybar

# Get Polybar Original Config
    $ sudo nvim /etc/polybar/config.ini
    // Copy to ~/.config/polybar/config.ini
    // And you can edit it

# Polybar Settings
[bar/mybar]
polybar mybar
