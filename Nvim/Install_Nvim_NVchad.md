# Install Neovim Widh NVchad
    // Install
        $ sudo pacman -S lua
        $ git clone https://github.com/NvChad/starter ~/.config/nvim && nvim

    // Uninstall
        // Linux / MacOS (unix)
            rm -rf ~/.config/nvim
            rm -rf ~/.local/state/nvim
            rm -rf ~/.local/share/nvim
    
    // Shortcut
    :Lazy sync
    <Space+t+h> --> change theems
    <Space+c+h> --> Chit Sheet
